program matrixtranspose;

main() {
    var int i, j, matrix[2][3], result[3][2];
        
    %% assign matrix
    matrix[0][0] = 1;
    matrix[1][0] = 2;
    matrix[0][1] = 3;
    matrix[1][1] = 4;
    matrix[0][2] = 5;
    matrix[1][2] = 6;

    print("Matrix assigned:");
    for j = 0 to j < 3 {
        for i = 0 to i < 2 {
            print(matrix[i][j]);
        }
    }

    result = matrix!;

    print("Result matrix:");
    for j = 0 to j < 2 {
        for i = 0 to i < 3 {
            print(result[i][j]);
        }
    }
}
